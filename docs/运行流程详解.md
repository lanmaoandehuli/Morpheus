```python
class DirectorAgent:
    async def generate_chapter_plan(self, project_id, chapter_title):
        # 1. 检索相关记忆
        memories = memory_store.query(
            project_id=project_id,
            query=f"第 1 章 {chapter_title}",
            top_k=10
        )
        
        # 2. 构建提示词
        prompt = self.build_planner_prompt(
            project_config=project.config,
            l1_identity=memories.l1,
            l2_entities=memories.l2,
            l3_events=memories.l3,
            chapter_title=chapter_title
        )
        
        # 3. 调用 GLM-4
        response = await llm_client.generate(
            model="glm-4-plus",
            prompt=prompt,
            temperature=0.7,
            max_tokens=2000
        )
        
        # 4. 解析输出
        plan = self.parse_plan(response.text)
        
        # 5. 保存计划
        self.save_plan(chapter_id, plan)
        
        return plan
```

```python
# backend/api/main.py
@app.get("/api/chapters/{chapter_id}/draft/stream")
async def stream_chapter_draft(chapter_id: str):
    async def generate():
        # 1. 加载章节计划
        plan = load_chapter_plan(chapter_id)
        
        # 2. 检索记忆
        memories = memory_store.query(
            project_id=project_id,
            query=plan.title,
            top_k=15
        )
        
        # 3. 构建写作提示词
        prompt = build_writer_prompt(
            plan=plan,
            style_guide=project.style,
            characters=memories.characters,
            previous_context=memories.events
        )
        
        # 4. 流式调用 GLM-4
        async for chunk in llm_client.stream_generate(
            model="glm-4-plus",
            prompt=prompt,
            temperature=0.8,
            max_tokens=4000
        ):
            # 5. 推送 SSE 事件
            yield f"data: {json.dumps({'type': 'chunk', 'text': chunk})}\n\n"
        
        # 6. 完成事件
        yield f"data: {json.dumps({'type': 'done'})}\n\n"
    
    return StreamingResponse(generate(), media_type="text/event-stream")
```

### 阶段 4: 多轮审查

#### 4.1 Setter Agent（设定审查）
```python
class SetterAgent:
    async def review_setting(self, draft, memories):
        # 检查世界观一致性
        issues = []
        
        # 检查 1: 时代背景
        if "手机" in draft and "末世后" in memories.events:
            if "通讯塔摧毁" in memories.events:
                issues.append("末世后不应有手机信号")
        
        # 检查 2: 角色能力
        if "飞行" in draft:
            if memories.characters["林辰"]["cultivation"] == "炼气期":
                issues.append("炼气期不能飞行")
        
        # 检查 3: 地点逻辑
        if "青阳市" in draft and "身在海外" in memories.location:
            issues.append("地点转换缺少交代")
        
        return {
            "passed": len(issues) == 0,
            "issues": issues,
            "suggestions": self.generate_suggestions(issues)
        }
```

#### 4.2 Stylist Agent（文笔润色）
```python
class StylistAgent:
    async def polish_style(self, draft):
        prompt = f"""
你是专业编辑，请润色以下小说段落：

要求:
1. 增强画面感
2. 优化对话节奏
3. 减少"的""了"重复
4. 增加感官描写

原文:
{draft}

请输出润色后的版本。
"""
        polished = await llm_client.generate(prompt)
        return polished
```

#### 4.3 Arbiter Agent（最终裁决）
```python
class ArbiterAgent:
    async def final_review(self, draft, all_reviews):
        # 综合所有审查意见
        major_issues = []
        minor_issues = []
        
        for review in all_reviews:
            if review.severity == "major":
                major_issues.append(review)
            else:
                minor_issues.append(review)
        
        # 裁决
        if len(major_issues) > 0:
            return {
                "status": "rejected",
                "reason": "存在重大设定冲突",
                "required_fixes": major_issues
            }
        else:
            return {
                "status": "approved",
                "suggestions": minor_issues
            }
```

## 🎯 关键设计亮点

### 1. 三层记忆系统
```
L1 Identity (核心设定)
   ↓ 指导
L2 Entities (实体信息)
   ↓ 参与
L3 Events (情节发展)
   ↓ 检索
```

```
Director (规划 + 写作)
   ↓
Setter (设定审查) → 不通过 → 返回修改
   ↓ 通过

   ↓
Arbiter (最终裁决) → 不通过 → 返回重写
   ↓ 通过
定稿 + 记忆更新
```

```
GLM-4 生成
   ↓ 分块

   ↓ 实时

   ↓ 逐字
```

```
   ↓ 记录

   ↓ 存储

   ↓ 可视化
```

```python
# 混合检索策略
def hybrid_search(query, top_k=30):
    # FTS 全文搜索 (快速)
    fts_results = fts_engine.search(query, top_k * 2)
    
    # 向量检索 (语义相似)
    vector_results = vector_engine.search(
        embedding_model.encode(query),
        top_k * 2
    )
    
    # Reciprocal Rank Fusion 合并
    fused_results = rrf_fusion(fts_results, vector_results)
    
    # 返回 top_k
    return fused_results[:top_k]
```
